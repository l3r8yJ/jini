  Jini.new(

        ''
      )
  )
end

def selection
  if @head.include?('[') || @head.include?(']') || @head.include?('@') || @head.include?('//')
    raise InvalidPath, 'Cannot select, path contains bad symbols'
  end
  Jini.new(@head.gsub('/', '::').to_s)
end
